### 文件命名

```
kebab-case:  task-socket.js ✅
camelCase:   taskSocket.js  ✅（JS 社区惯例，两者均可，项目内统一即可）
PascalCase:  TaskSocket.js  ❌（非类文件）
```

本项目统一用 **kebab-case**（与目录结构一致）。

### 函数命名

```js
// ✅ camelCase，动词开头

export function watchTask({ taskId, onUpdate, onError }) { ... }
export function openTaskChannel(taskId, token, onMessage, onError) { ... }

// ❌ 避免模糊命名
export function task() { ... }   // ❌
export function doIt() { ... }   // ❌
```

## 异步规范

```js
// ✅ async/await 替代 .then/.catch 链
async function loadWork(id) {
  try {
    const work = await getWork(id);
    renderWork(work);
  } catch (err) {
    showError(err.message);
  }
}

// ❌ 回调地狱
getWork(id, function(work) {
  renderWork(work, function() { ... });  // ❌
});

// ✅ 并发请求用 Promise.all
const [works, balance] = await Promise.all([
  getWorks(),
  getBalance(),
]);
```

## 错误处理规范

```js
// ✅ 自定义错误类
export class APIError extends Error {
  constructor(code, message, status) {
    super(message);
    this.name = 'APIError';
    this.code = code;
    this.status = status;
  }
}

// ✅ client.js 统一抛出 APIError
async function request(path, options) {
  const resp = await fetch(baseUrl + path, options);
  if (!resp.ok) {
    const body = await resp.json().catch(() => ({}));
    throw new APIError(body.code ?? 'UNKNOWN', body.message ?? 'Request failed', resp.status);
  }
  return resp.json();
}

// ✅ 调用层捕获具体错误类型
try {
  await createTask(prompt);
} catch (err) {
  if (err instanceof APIError && err.code === 'RATE_LIMIT_EXCEEDED') {
    showRateLimitHint();
  } else {
    showGenericError(err.message);
  }
}
```

## Token 刷新规范

```js
// ✅ 刷新只更新 AT，不覆盖 RT（TokenPair 规范 §3.1）
async function refresh() {
  const rt = localStorage.getItem('calliope_rt');
  const resp = await fetch(`${BASE_URL}/auth/refresh`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ refreshToken: rt }),
  });
  if (!resp.ok) throw new Error('REFRESH_FAILED');
  const { accessToken, expiresAt } = await resp.json();
  localStorage.setItem('calliope_at', accessToken);
  localStorage.setItem('calliope_at_expires', String(new Date(expiresAt).getTime()));
  // ❌ 不写 refreshToken：后端不轮换 RT
}
```

## 测试规范

```js
// ✅ Vitest 单元测试
import { describe, it, expect, vi } from 'vitest';
import { openTaskChannel } from '../src/ws/task-socket.js';

describe('openTaskChannel', () => {
  it('calls onError when WS onerror fires', () => {
    const onError = vi.fn();
    const cancel = openTaskChannel('task-1', 'token', vi.fn(), onError);
    // 模拟 WS error 事件...
    expect(onError).toHaveBeenCalled();
    cancel();
  });
});

// ✅ fetch mock
vi.stubGlobal('fetch', vi.fn().mockResolvedValue({
  ok: true,
  json: async () => ({ taskId: 'abc' }),
}));
```

- `ws/task-socket.js` 100%（WS 状态机）
- `audio/player.js` > 70%
- `storage/token-store.js` 100%（token 读写和过期逻辑）

## Safari 兼容注意事项

```js
// ✅ audio.play() 必须在用户手势回调中调用（Safari 限制）
playButton.addEventListener('click', async () => {
  await player.play();  // ✅ 在事件回调中
});

// ❌ 不能在 Promise 回调外部延迟调用
setTimeout(() => player.play(), 100);  // ❌ Safari 会阻止
```
